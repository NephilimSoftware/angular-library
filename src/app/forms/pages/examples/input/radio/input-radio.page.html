Radio buttons

Values can be simple types or user defined classes

public readonly userDefinedValues: UserDefinedClass[] = [
new UserDefinedClass('A'),
new UserDefinedClass('B'),
new UserDefinedClass('C'),
new UserDefinedClass('D'), ];
public readonly firkrag: FormControl<UserDefinedClass> = new FormControl<UserDefinedClass>(this.userDefinedValues[1]);

Additional definitions used in example:

class UserDefinedClass {
  constructor(public readonly label: string) {}

  public toString(): string {
    return `[UserDefinedClass label: ${this.label}]`;
  }
}
<label *ngFor="let value of userDefinedValues">
<input [formControl]="firkrag" [value]="value" name="test" type="radio" /> {{ '{{' }} value.label }} </label>
